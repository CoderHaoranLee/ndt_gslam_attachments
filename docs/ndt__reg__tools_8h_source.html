146 double trialValueSelectionMT(double a_l, double f_l, double g_l, double a_u,
147  double f_u, double g_u, double a_t, double f_t,
148  double g_t)
149 {
150  // Case 1 in Trial Value Selection [More, Thuente 1994]
151  if (f_t > f_l) {
152  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l
153  // and
154  // g_t
155  // Equation 2.4.52 [Sun, Yuan 2006]
156  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
157  double w = std::sqrt(z * z - g_t * g_l);
158  // Equation 2.4.56 [Sun, Yuan 2006]
159  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
160 
161  // Calculate the minimizer of the quadratic that interpolates f_l, f_t and
162  // g_l
163  // Equation 2.4.2 [Sun, Yuan 2006]
164  double a_q =
165  a_l - 0.5 * (a_l - a_t) * g_l / (g_l - (f_l - f_t) / (a_l - a_t));
166 
167  if (std::fabs(a_c - a_l) < std::fabs(a_q - a_l))
168  return (a_c);
169  else
170  return (0.5 * (a_q + a_c));
171  }
172  // Case 2 in Trial Value Selection [More, Thuente 1994]
173  else if (g_t * g_l < 0) {
174  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l
175  // and
176  // g_t
177  // Equation 2.4.52 [Sun, Yuan 2006]
178  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
179  double w = std::sqrt(z * z - g_t * g_l);
180  // Equation 2.4.56 [Sun, Yuan 2006]
181  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
182 
183  // Calculate the minimizer of the quadratic that interpolates f_l, g_l and
184  // g_t
185  // Equation 2.4.5 [Sun, Yuan 2006]
186  double a_s = a_l - (a_l - a_t) / (g_l - g_t) * g_l;
187 
188  if (std::fabs(a_c - a_t) >= std::fabs(a_s - a_t))
189  return (a_c);
190  else
191  return (a_s);
192  }
193  // Case 3 in Trial Value Selection [More, Thuente 1994]
194  else if (std::fabs(g_t) <= std::fabs(g_l)) {
195  // Calculate the minimizer of the cubic that interpolates f_l, f_t, g_l
196  // and
197  // g_t
198  // Equation 2.4.52 [Sun, Yuan 2006]
199  double z = 3 * (f_t - f_l) / (a_t - a_l) - g_t - g_l;
200  double w = std::sqrt(z * z - g_t * g_l);
201  double a_c = a_l + (a_t - a_l) * (w - g_l - z) / (g_t - g_l + 2 * w);
202 
203  // Calculate the minimizer of the quadratic that interpolates g_l and g_t
204  // Equation 2.4.5 [Sun, Yuan 2006]
205  double a_s = a_l - (a_l - a_t) / (g_l - g_t) * g_l;
206 
207  double a_t_next;
208 
209  if (std::fabs(a_c - a_t) < std::fabs(a_s - a_t))
210  a_t_next = a_c;
211  else
212  a_t_next = a_s;
213 
214  if (a_t > a_l)
215  return (std::min(a_t + 0.66 * (a_u - a_t), a_t_next));
216  else
217  return (std::max(a_t + 0.66 * (a_u - a_t), a_t_next));
218  }
219  // Case 4 in Trial Value Selection [More, Thuente 1994]
220  else {
221  // Calculate the minimizer of the cubic that interpolates f_u, f_t, g_u
222  // and
223  // g_t
224  // Equation 2.4.52 [Sun, Yuan 2006]
225  double z = 3 * (f_t - f_u) / (a_t - a_u) - g_t - g_u;
226  double w = std::sqrt(z * z - g_t * g_u);
227  // Equation 2.4.56 [Sun, Yuan 2006]
228  return (a_u + (a_t - a_u) * (w - g_u - z) / (g_t - g_u + 2 * w));
229  }
