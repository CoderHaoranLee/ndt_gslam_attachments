836 template <typename CellType>
837 bool VoxelGrid2D<CellType>::linesIntersection(const Eigen::Vector2d& p,
838  const Eigen::Vector2d& p2,
839  const Eigen::Vector2d& q,
840  const Eigen::Vector2d& q2,
841  Eigen::Vector2d& intersection,
842  bool output_colinear) const
843 {
844  // Eigen::Vector3d pw;
845  // Eigen::Vector3d p2w;
846  // Eigen::Vector3d qw;
847  // Eigen::Vector3d q2w;
848  // pw << p(0), p(1), 0;
849  // p2w << p2(0), p2(1), 0;
850  // qw << q(0), q(1), 0;
851  // q2w << q2(0), q2(1), 0;
852  double epsilon = 1e-7;
853  intersection.setZero();
854  Eigen::Vector2d r = p2 - p;
855  Eigen::Vector2d s = q2 - q;
856  double rxs = cross(r, s);
857  double qpxr = cross((q - p), r);
858  // If r x s = 0 and (q - p) x r = 0, then the two lines are collinear.
859  if (std::abs(rxs) < epsilon && std::abs(qpxr) < epsilon) {
860  // 1. If either 0 <= (q - p) * r <= r * r or 0 <= (p - q) * s <= * s
861  // then the two lines are overlapping,
862  if (output_colinear)
863  if ((0 <= (q - p).dot(r) && (q - p).dot(r) <= r.dot(r)) ||
864  (0 <= (p - q).dot(s) && (p - q).dot(s) <= s.dot(s)))
865  return true;
866 
867  // 2. If neither 0 <= (q - p) * r = r * r nor 0 <= (p - q) * s <= s * s
868  // then the two lines are collinear but disjoint.
869  // No need to implement this expression, as it follows from the expression
870  // above.
871  return false;
872  }
873 
874  // 3. If r x s = 0 and (q - p) x r != 0, then the two lines are parallel and
875  // non-intersecting.
876  if (std::abs(rxs) < epsilon && std::abs(qpxr) >= epsilon)
877  return false;
878 
879  // t = (q - p) x s / (r x s)
880  double t = cross((q - p), s) / rxs;
881 
882  // u = (q - p) x r / (r x s)
883 
884  double u = cross((q - p), r) / rxs;
885 
886  // 4. If r x s != 0 and 0 <= t <= 1 and 0 <= u <= 1
887  // the two line segments meet at the point p + t r = q + u s.
888  if (std::abs(rxs) >= epsilon && (0 <= t && t <= 1) && (0 <= u && u <= 1)) {
889  // We can calculate the intersection point using either t or u.
890  intersection = p + t * r;
891 
892  // An intersection was found.
893  return true;
894  }
895 
896  // 5. Otherwise, the two line segments are not parallel but do not
897  // intersect.
898  return false;
899 }
