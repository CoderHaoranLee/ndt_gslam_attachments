932 template <typename CellType, typename PointType>
933 typename NDTGrid2D<CellType, PointType>::CellPtrVector
934 NDTGrid2D<CellType, PointType>::getKNearestNeighborsVoxel(
935  const Eigen::Vector2d &pt, size_t K) const
936 {
937  Eigen::Vector3d point(pt(0), pt(1), 0);
938  CellPtrVector res;
939  // slowly increase search radius and find sufficient number of points
940  CellPtrVector radius_cells = getNeighbors(pt, 1);
941  size_t radius = 2;
942  while (radius_cells.size() < K && radius < grid_.width()) {
943  radius_cells = getNeighbors(pt, radius);
944  radius = radius * 2;
945  }
946  // not enough elements in grid
947  if (radius_cells.size() < K)
948  return radius_cells;
949  // sort elements based on distance from point from parameter
950  std::sort(radius_cells.begin(), radius_cells.end(), [&point](CellType *a,
951  CellType *b) {
952  return (a->getMean() - point).norm() < (a->getMean() - point).norm();
953  });
954  res.reserve(K);
955  std::move(radius_cells.begin(), radius_cells.begin() + K,
956  std::back_inserter(res));
957  return res;
958 }
