317  computeSingleGrid(PclSource &output, const GridTarget &target_grid,
318  const Eigen::Matrix4f &guess,
319  const ndt_reg::FittingParams &param)
320 {

322  converged_ = false;
323  // Initialise final transformation to the guessed one
324  final_transformation_ = guess;
325  previous_transformation_ = final_transformation_;
326  // variables needed for calculation
327  Eigen::Vector3d xytheta_p = ndt_reg::matToVec(guess);
328  Eigen::Vector3d delta_xytheta_p;
329  Eigen::Matrix4f p;
330  double delta_p_norm;
331  ndt_reg::ScoreAndDerivatives<> score;
332  while (!converged_) {
333  score = calcScore(target_grid, *input_, xytheta_p, param);
334  // Solve for decent direction using newton method, line 23 in Algorithm 2
335  // [Magnusson 2009]
336  Eigen::JacobiSVD<Eigen::Matrix3d> sv(
337  score.hessian_, Eigen::ComputeFullU | Eigen::ComputeFullV);
338  // Negative for maximization as opposed to minimization
339  delta_xytheta_p = sv.solve(-score.gradient_);
340 
341  // Calculate step length with guarnteed sufficient decrease [More, Thuente
342  // 1994]
343  delta_p_norm = delta_xytheta_p.norm();
344  if (delta_p_norm == 0 || delta_p_norm != delta_p_norm) {
345  trans_probability_ =
346  score.value_ / static_cast<double>(input_->points.size());
347  converged_ = delta_p_norm == delta_p_norm;
348  covariance_.setIdentity();
349  inform_matrix_.setIdentity();
350  ROS_ERROR_STREAM(

352  return false;
353  }
354  delta_xytheta_p.normalize();
355  delta_p_norm = computeStepLengthMT(xytheta_p, delta_xytheta_p, delta_p_norm,
356  step_size_, transformation_epsilon_ / 2,
357  target_grid, score, output, param);
358  // delta_p_norm = 0.8
359  delta_xytheta_p *= delta_p_norm;
360  xytheta_p += delta_xytheta_p;
361  p = ndt_reg::vecToMat<float>(xytheta_p);
362 
363  // convergence testing
364  if (nr_iterations_ > max_iterations_ ||
365  (nr_iterations_ &&
366  (std::abs(delta_p_norm) < transformation_epsilon_))) {
367  converged_ = true;
368  }
369 
370  ++nr_iterations_;
371  previous_transformation_ = transformation_;
372  transformation_ = p;
373  trans_probability_ =
374  score.value_ / static_cast<double>(input_->points.size());
375  ROS_DEBUG_STREAM("[NDT2D]: Step: "
376  << delta_p_norm
377  << " Delta: " << delta_xytheta_p.transpose()

386  }

388  transformPointCloud(*input_, output, transformation_);
389  covariance_ = score.hessian_;
390  inform_matrix_ = score.hessian_.inverse();

396 ndt_reg::ScoreAndDerivatives<3, double>
397 NormalDistributionsTransform2DEx<PointSource, PointTarget, CellType>::calcScore(
398  const GridTarget &ndt_grid, const PclSource &source,
399  const Eigen::Vector3d &trans, const ndt_reg::FittingParams &param)
400 {
401  PclSource trans_source;
402  ndt_reg::ScoreAndDerivatives<3, double> res;
403  double cos_theta, sin_theta;
404  // prepare angles for jaccobian and hessian derivatives usage
405  // simplify small angles
406  if (std::abs(trans(2)) < 10e-5) {
407  cos_theta = 1.0;
408  sin_theta = 0.0;
409  } else {
410  cos_theta = std::cos(trans(2));
411  sin_theta = std::sin(trans(2));
412  }
413  res = ndt_reg::ScoreAndDerivatives<3, double>::Zero();
414  transformPointCloud(source, trans_source, ndt_reg::vecToMat(trans));
415 
416  // calculate score for each point
417  typedef typename GridTarget::CellPtrVector TargetVec;
418  for (size_t i = 0; i < trans_source.size(); ++i) {
419  Eigen::Vector2d pt(trans_source[i].x, trans_source[i].y);
420 
421  TargetVec neighbors = ndt_grid.getKNearestNeighbors(pt, 4);
422 
423  for (size_t nb = 0; nb < neighbors.size(); nb++) {
424  res += calcPointScore(source[i], neighbors[nb], trans_source[i],
425  cos_theta, sin_theta, param);
426  }
427  // ROS_DEBUG_STREAM("Number of neighbours for point: "<<count);
428  }

433 template <typename PointSource, typename PointTarget, typename CellType>
434 ndt_reg::ScoreAndDerivatives<3, double> NormalDistributionsTransform2DEx<
435  PointSource, PointTarget,
436  CellType>::calcPointScore(const PointSource &pt, const CellType *cell,
437  const PointSource &trans_pt, double cos_theta,
438  double sin_theta,
439  const ndt_reg::FittingParams &param) const
440 {
441  ndt_reg::ScoreAndDerivatives<3, double> res;
442  Eigen::Matrix<double, 2, 3> jacobian;
443  Eigen::Vector2d hess_derivative, hess_derivatives2;
444  double x = pt.x;
445  double y = pt.y;
446  double co = cos_theta;
447  double si = sin_theta;
448 
449  jacobian << 1, 0, -x * si - y * co, 0, 1, x * co - y * si;
450  hess_derivative << -x * co + y * si, -x * si - y * co;
451  hess_derivatives2 << 0, 0;
452 
453  Eigen::Matrix2d icov = cell->getICov().block(0, 0, 2, 2);
454  Eigen::Vector2d mean = cell->getMean().head(2);
455  Eigen::Vector2d trans_point(trans_pt.x, trans_pt.y);
456  Eigen::Vector2d pt_mean_diff = trans_point - mean;
457  Eigen::Vector2d diff_icov = pt_mean_diff.transpose() * icov;
458 
459  double e_x_cov_x =
460  std::exp(-param.gauss_d2__half_ * diff_icov.dot(pt_mean_diff));
461  double d2_e_x_cov_x = param.gauss_d2_ * e_x_cov_x;
462  double d1d2_e_x_cov_x = param.gauss_d1_ * d2_e_x_cov_x;
463 
464  if (d2_e_x_cov_x > 1 || d2_e_x_cov_x < 0 || d2_e_x_cov_x != d2_e_x_cov_x) {
465  ROS_DEBUG_STREAM("[NDT2D]:Wrong probability density function (PDF)");
466  return res.Zero();
467  }
468  // calculate score (eq. 6.9) [Magnusson 2009]
469  res.value_ = -param.gauss_d1_ * e_x_cov_x;
470 
471  Eigen::Vector2d hess_part;
472  for (long i = 0; i < 3; ++i) {
473  // calculate gradient (eq. 6.12) [Magnusson 2009]
474  res.gradient_(i) = diff_icov.dot(jacobian.col(i)) * d1d2_e_x_cov_x;
475  for (long j = 0; j < 3; ++j) {
476  // calculate hessian (eq. 6.13) [Magnusson 2009]
477  if (i == 2 && j == 2) {
478  hess_part = hess_derivatives2;
479  } else {
480  hess_part = hess_derivative;
481  }
482  double jacc_i = diff_icov.dot(jacobian.col(i));
483  double jacc_j = diff_icov.dot(jacobian.col(j));
484  res.hessian_(i, j) =
485  d1d2_e_x_cov_x *
486  (-param.gauss_d2_ * jacc_i * jacc_j + diff_icov.dot(hess_part) +
487  jacobian.col(j).dot(icov * jacobian.col(i)));
488  }
489  }
490  return res;
491 }

494 template <typename PointSource, typename PointTarget, typename CellType>
495 double NormalDistributionsTransform2DEx<PointSource, PointTarget, CellType>::
496  computeStepLengthMT(const Eigen::Matrix<double, 3, 1> &x,
497  Eigen::Matrix<double, 3, 1> &step_dir, double step_init,
498  double step_max, double step_min,
499  const GridTarget &ndt_grid,
500  const ndt_reg::ScoreAndDerivatives<3, double> &score,
501  const PclSource &input,
502  const ndt_reg::FittingParams &param)
503 {
504  Eigen::Matrix4f transformation;
505  PclSource trans_cloud;
506  // transformation.setIdentity();
507  ndt_reg::ScoreAndDerivatives<3, double> score_vals = score;
508  // Set the value of phi(0), Equation 1.3 [More, Thuente 1994]
509  double phi_0 = -score_vals.value_;
510  // Set the value of phi'(0), Equation 1.3 [More, Thuente 1994]
511  double d_phi_0 = -(score_vals.gradient_.dot(step_dir));
512 
513  Eigen::Matrix<double, 3, 1> x_t;
514 
515  if (d_phi_0 >= 0) {
516  // Not a decent direction
517  if (d_phi_0 == 0)
518  return 0;
519  else {
520  // Reverse step direction and calculate optimal step.
521  d_phi_0 *= -1;
522  step_dir *= -1;
523  }
524  }
525 
526  // The Search Algorithm for T(mu) [More, Thuente 1994]
527 
528  int max_step_iterations = 10;
529  int step_iterations = 0;
530 
531  // Sufficient decreace constant, Equation 1.1 [More, Thuete 1994]
532  double mu = 1.e-4;
533  // Curvature condition constant, Equation 1.2 [More, Thuete 1994]
534  double nu = 0.9;
535 
536  // Initial endpoints of Interval I,
537  double a_l = 0, a_u = 0;
538 
539  // Auxiliary function psi is used until I is determined ot be a closed
540  // interval, Equation 2.1 [More, Thuente 1994]
541  double f_l = ndt_reg::auxilaryFunction_PsiMT(a_l, phi_0, phi_0, d_phi_0, mu);
542  double g_l = ndt_reg::auxilaryFunction_dPsiMT(d_phi_0, d_phi_0, mu);
543 
544  double f_u = ndt_reg::auxilaryFunction_PsiMT(a_u, phi_0, phi_0, d_phi_0, mu);
545  double g_u = ndt_reg::auxilaryFunction_dPsiMT(d_phi_0, d_phi_0, mu);
546 
547  // Check used to allow More-Thuente step length calculation to be skipped by
548  // making step_min == step_max
549  bool interval_converged = (step_max - step_min) > 0, open_interval = true;
550 
551  double a_t = step_init;
552  a_t = std::min(a_t, step_max);
553  a_t = std::max(a_t, step_min);
554 
555  x_t = x + step_dir * a_t;
556 
557  // transformation = ndt_reg::vecToMat(x_t);
558 
559  // Updates score, gradient and hessian. Hessian calculation is unessisary but
560  // testing showed that most step calculations use the
561  // initial step suggestion and recalculation the reusable portions of the
562  // hessian would intail more computation time.
563  score_vals = calcScore(ndt_grid, input, x_t, param);
564 
565  // Calculate phi(alpha_t)
566  double phi_t = -score_vals.value_;
567  // Calculate phi'(alpha_t)
568  double d_phi_t = -(score_vals.gradient_.dot(step_dir));
569 
570  // Calculate psi(alpha_t)
571  double psi_t =
572  ndt_reg::auxilaryFunction_PsiMT(a_t, phi_t, phi_0, d_phi_0, mu);
573  // Calculate psi'(alpha_t)
574  double d_psi_t = ndt_reg::auxilaryFunction_dPsiMT(d_phi_t, d_phi_0, mu);
575 
576  // Iterate until max number of iterations, interval convergance or a value
577  // satisfies the sufficient decrease, Equation 1.1, and curvature condition,
578  // Equation 1.2 [More, Thuente 1994]
579  while (!interval_converged && step_iterations < max_step_iterations &&
580  !(psi_t <= 0 /*Sufficient Decrease*/ &&
581  d_phi_t <= -nu * d_phi_0 /*Curvature Condition*/)) {
582  // Use auxilary function if interval I is not closed
583  if (open_interval) {
584  a_t = ndt_reg::trialValueSelectionMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t,
585  psi_t, d_psi_t);
586  } else {
587  a_t = ndt_reg::trialValueSelectionMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t,
588  phi_t, d_phi_t);
589  }
590 
591  a_t = std::min(a_t, step_max);
592  a_t = std::max(a_t, step_min);
593 
594  x_t = x + step_dir * a_t;
595  // transformation = ndt_reg::vecToMat(x_t);
596 
597  score_vals = calcScore(ndt_grid, input, x_t, param);
598 
599  // Calculate phi(alpha_t+)
600  phi_t = -score_vals.value_;
601  // Calculate phi'(alpha_t+)
602  d_phi_t = -(score_vals.gradient_.dot(step_dir));
603 
604  // Calculate psi(alpha_t+)
605  psi_t = ndt_reg::auxilaryFunction_PsiMT(a_t, phi_t, phi_0, d_phi_0, mu);
606  // Calculate psi'(alpha_t+)
607  d_psi_t = ndt_reg::auxilaryFunction_dPsiMT(d_phi_t, d_phi_0, mu);
608 
609  // Check if I is now a closed interval
610  if (open_interval && (psi_t <= 0 && d_psi_t >= 0)) {
611  open_interval = false;
612 
613  // Converts f_l and g_l from psi to phi
614  f_l = f_l + phi_0 - mu * d_phi_0 * a_l;
615  g_l = g_l + mu * d_phi_0;
616 
617  // Converts f_u and g_u from psi to phi
618  f_u = f_u + phi_0 - mu * d_phi_0 * a_u;
619  g_u = g_u + mu * d_phi_0;
620  }
621 
622  if (open_interval) {
623  // Update interval end points using Updating Algorithm [More, Thuente
624  // 1994]
625  interval_converged = ndt_reg::updateIntervalMT(a_l, f_l, g_l, a_u, f_u,
626  g_u, a_t, psi_t, d_psi_t);
627  } else {
628  // Update interval end points using Modified Updating Algorithm [More,
629  // Thuente 1994]
630  interval_converged = ndt_reg::updateIntervalMT(a_l, f_l, g_l, a_u, f_u,
631  g_u, a_t, phi_t, d_phi_t);
632  }
633 
634  step_iterations++;
635  }
636 
637  // If inner loop was run then hessian needs to be calculated.
638  // Hessian is unnessisary for step length determination but gradients are
639  // required
640  // so derivative and transform data is stored for the next iteration.
641  // if (step_iterations)
642  // computeHessian (hessian, trans_cloud, x_t);
643 
644  return (a_t);
645 }
