197  theta += coarse_rot_step_) {

199  }

202  // time:");

207  std::vector<ml_corr::IndexPoint> transl_index_pcl; // translated rotated

212  for (size_t theta_id = 0; theta_id < rotations.size(); ++theta_id) {
213  // prepare storage for calculated voxels
214  std::vector<ml_corr::SearchVoxel> local_voxels;
215  local_voxels.reserve(static_cast<size_t>(
216  std::ceil(std::pow(2 * translation_range_ / coarse_step_, 2))));
217  // rotate point cloud

219  index_pcl = coarse_lookup_.toIndexes(rot_pcl);
220  // iterate over all possible translation x values

222  dx += coarse_step_) {
223  // iterate over all possible y translation values

228  voxel.transform_ << dx, dy, rotations[theta_id];
229  voxel.score_ = coarse_lookup_.getScore(transl_index_pcl);
230  local_voxels.push_back(std::move(voxel));
231  }
232  }

234  std::move(local_voxels.begin(), local_voxels.end(),
235  std::back_inserter(search_voxels_thread[thread_id]));
236  }
237 
238  // end of parallel world

242  std::back_inserter(search_voxels));
243  }
244  } // end of measuring scope
245  // find voxel with best score = best chance for good match
246  std::sort(search_voxels.begin(), search_voxels.end());
247  ml_corr::SearchVoxel best_voxel;
248

250  ROS_DEBUG_STREAM(

252 
253  // output data

256  transformPointCloud(*input_, output, final_transformation_);
