127 bool EdgeSE3Mixture::read(std::istream& is)
128 {
129  is >> numberComponents;
130 
131  allEdges.reserve(numberComponents);
132  weights.reserve(numberComponents);
133  determinants.reserve(numberComponents);
134 
135  Vector7d p;
136  double w;
137 
138  VertexSE3* va = static_cast<VertexSE3*>(this->vertex(0));
139 
140  for (int c = 0; c < numberComponents; c++) {
141  EdgeSE3* e = new EdgeSE3;
142  allEdges.push_back(e);
143  std::string buf;
144  is >> buf;
145  is >> w;
146  weights.push_back(w);
147  int na, nb;
148  is >> na;
149  is >> nb;
150 
151  VertexSE3* v0 = static_cast<VertexSE3*>(va->graph()->vertex(na));
152  VertexSE3* v1 = static_cast<VertexSE3*>(va->graph()->vertex(nb));
153  assert(v0 != NULL);
154  assert(v1 != NULL);
155  allEdges[c]->setVertex(0, v0);
156  allEdges[c]->setVertex(1, v1);
157 
158  for (int i = 0; i < 7; i++)
159  is >> p[i];
160  Vector4d::MapType(p.data() + 3).normalize();
161 
162  allEdges[c]->setMeasurementData(p.data());
163  InformationType inf;
164  for (int i = 0; i < 6; ++i)
165  for (int j = i; j < 6; ++j) {
166  is >> inf(i, j);
167  if (i != j)
168  inf(j, i) = inf(i, j);
169  }
170  allEdges[c]->setInformation(inf);
171  }
172 
173  for (unsigned int i = 0; i < numberComponents; i++)
174  determinants.push_back(allEdges[i]->information().inverse().determinant());
175 
176  computeBestEdge();
177  return is.good();
178 }
