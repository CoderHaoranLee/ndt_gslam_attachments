584 void D2DNormalDistributionsTransform2D<PointSource, PointTarget, CellType>::
585  computeDerivatives(const Eigen::Vector3d &x, const Eigen::Matrix3d &cov,
586  ndt_reg::JacobianHessianDerivatives &data,
587  bool calc_hessian)
588 {
589  data.setZero();

592  data.Jest(1, 2) = x(0);
593 
594  //_Zest

597  cov(0, 2), 0;

602  -4 * cov(0, 1), -cov(0, 2), -4 * cov(0, 1),

607 template <typename PointSource, typename PointTarget, typename CellType>
608 ndt_reg::ScoreAndDerivatives<3, double>
609 D2DNormalDistributionsTransform2D<PointSource, PointTarget, CellType>::
610  calcSourceCellScore(const Eigen::Vector3d &mean_source,
611  const Eigen::Matrix3d &cov_source,
612  const CellType *cell_t,
613  const ndt_reg::JacobianHessianDerivatives &deriv,
614  const ndt_reg::FittingParams &param, bool calc_hessian)
615 {
616  ndt_reg::ScoreAndDerivatives<3, double> res;
617 
618  // declaration
619  Eigen::Vector3d diff_mean;
620  Eigen::Matrix3d cov_sum, icov;
621  cov_sum.setIdentity();
622  double det = 0;
623  bool exists = false;
624  double dist;
625  // vars for gradient
626  Eigen::Matrix<double, 3, 1> xtBJ, xtBZBx, Q;
627  // vars for hessian
628  Eigen::Matrix<double, 3, 3> xtBZBJ, xtBH, xtBZBZBx, xtBZhBx;
629  Eigen::Matrix<double, 1, 3> TMP1, xtB;
630 
631  xtBJ.setZero();
632  xtBZBx.setZero();
633  Q.setZero();
634  xtBZBJ.setZero();
635  xtBH.setZero();
636  xtBZBZBx.setZero();
637  xtBZhBx.setZero();
638  TMP1.setZero();
639  xtB.setZero();
640 
641  diff_mean = (mean_source - cell_t->getMean());
642  cov_sum = (cell_t->getCov() + cov_source);
643 
644  cov_sum.computeInverseAndDetWithCheck(icov, det, exists);
645  if (!exists) {
646  return res.Zero();
647  }
648  dist = (diff_mean).dot(icov * (diff_mean));
649  if (dist * 0 != 0) {
650  return res.Zero();
651  }
652  res.value_ = -param.gauss_d1_ * std::exp(-param.gauss_d2__half_ * dist);
653 
654  xtB = diff_mean.transpose() * icov;
655  xtBJ = xtB * deriv.Jest;
656 
657  TMP1 = xtB * deriv.Zest.block<3, 3>(0, 6) * icov;
658  xtBZBx(2) = TMP1 * diff_mean;
659  if (calc_hessian) {
660  xtBZBJ.col(2) = (TMP1 * deriv.Jest).transpose();
661  for (unsigned int j = 0; j < 3; j++) {
662  xtBH(2, j) = xtB * deriv.Hest.block<3, 1>(6, j);
663  xtBZBZBx(2, j) =
664  TMP1 * deriv.Zest.block<3, 3>(0, 3 * j) * icov * diff_mean;
665  xtBZhBx(2, j) =
666  xtB * deriv.ZHest.block<3, 3>(6, 3 * j) * icov * diff_mean;
667  }
668  }
669  Q = 2 * xtBJ - xtBZBx;
670  double factor = -(param.gauss_d2__half_) * res.value_;
671  res.gradient_ += Q * factor;
672 
673  if (calc_hessian) {
674  res.hessian_ += factor * (2 * deriv.Jest.transpose() * icov * deriv.Jest +
675  2 * xtBH - xtBZhBx - 2 * xtBZBJ.transpose() -
676  2 * xtBZBJ + xtBZBZBx + xtBZBZBx.transpose() -
677  param.gauss_d2__half_ * Q * Q.transpose());
678  }
679 
680  return res;
681 }

732  double g_l = ndt_reg::auxilaryFunction_dPsiMT(d_phi_0, d_phi_0, mu);
733 
734  double f_u = ndt_reg::auxilaryFunction_PsiMT(a_u, phi_0, phi_0, d_phi_0, mu);
735  double g_u = ndt_reg::auxilaryFunction_dPsiMT(d_phi_0, d_phi_0, mu);
736 
737  // Check used to allow More-Thuente step length calculation to be skipped by
738  // making step_min == step_max
739  bool interval_converged = (step_max - step_min) > 0, open_interval = true;
740 
741  double a_t = step_init;
742  a_t = std::min(a_t, step_max);
743  a_t = std::max(a_t, step_min);
744 
745  x_t = x + step_dir * a_t;
746 
747  // transformation = ndt_reg::vecToMat(x_t);
748 
749  // Updates score, gradient and hessian. Hessian calculation is unessisary
750  // but
751  // testing showed that most step calculations use the
752  // initial step suggestion and recalculation the reusable portions of the
753  // hessian would intail more computation time.
754  score_vals = calcScore(param, source_grid, x_t, target_grid, false);
755 
756  // Calculate phi(alpha_t)
757  double phi_t = -score_vals.value_;
758  // Calculate phi'(alpha_t)
759  double d_phi_t = -(score_vals.gradient_.dot(step_dir));
760 
761  // Calculate psi(alpha_t)
762  double psi_t =
763  ndt_reg::auxilaryFunction_PsiMT(a_t, phi_t, phi_0, d_phi_0, mu);
764  // Calculate psi'(alpha_t)
765  double d_psi_t = ndt_reg::auxilaryFunction_dPsiMT(d_phi_t, d_phi_0, mu);
766 
767  // Iterate until max number of iterations, interval convergance or a value
768  // satisfies the sufficient decrease, Equation 1.1, and curvature condition,
769  // Equation 1.2 [More, Thuente 1994]
770  while (!interval_converged && step_iterations < max_step_iterations &&
771  !(psi_t <= 0 /*Sufficient Decrease*/ &&
772  d_phi_t <= -nu * d_phi_0 /*Curvature Condition*/)) {
773  // Use auxilary function if interval I is not closed
774  if (open_interval) {
775  a_t = ndt_reg::trialValueSelectionMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t,
776  psi_t, d_psi_t);
777  } else {
778  a_t = ndt_reg::trialValueSelectionMT(a_l, f_l, g_l, a_u, f_u, g_u, a_t,
779  phi_t, d_phi_t);
780  }
781 
782  a_t = std::min(a_t, step_max);
783  a_t = std::max(a_t, step_min);
784 
785  x_t = x + step_dir * a_t;
786  // transformation = ndt_reg::vecToMat(x_t);
787 
788  score_vals = calcScore(param, source_grid, x_t, target_grid, false);
789 
790  // Calculate phi(alpha_t+)
791  phi_t = -score_vals.value_;
792  // Calculate phi'(alpha_t+)
793  d_phi_t = -(score_vals.gradient_.dot(step_dir));
794 
795  // Calculate psi(alpha_t+)
796  psi_t = ndt_reg::auxilaryFunction_PsiMT(a_t, phi_t, phi_0, d_phi_0, mu);
797  // Calculate psi'(alpha_t+)
798  d_psi_t = ndt_reg::auxilaryFunction_dPsiMT(d_phi_t, d_phi_0, mu);
799 
800  // Check if I is now a closed interval
801  if (open_interval && (psi_t <= 0 && d_psi_t >= 0)) {
802  open_interval = false;
803 
804  // Converts f_l and g_l from psi to phi
805  f_l = f_l + phi_0 - mu * d_phi_0 * a_l;
806  g_l = g_l + mu * d_phi_0;
807 
808  // Converts f_u and g_u from psi to phi
809  f_u = f_u + phi_0 - mu * d_phi_0 * a_u;
810  g_u = g_u + mu * d_phi_0;
811  }
812 
813  if (open_interval) {
814  // Update interval end points using Updating Algorithm [More, Thuente
815  // 1994]
816  interval_converged = ndt_reg::updateIntervalMT(a_l, f_l, g_l, a_u, f_u,
817  g_u, a_t, psi_t, d_psi_t);
818  } else {
819  // Update interval end points using Modified Updating Algorithm [More,
820  // Thuente 1994]
821  interval_converged = ndt_reg::updateIntervalMT(a_l, f_l, g_l, a_u, f_u,
822  g_u, a_t, phi_t, d_phi_t);
823  }
824 
825  step_iterations++;
826  }
827  return (a_t);
828 }
829 } // end of pcl namespace
