62 void EdgeSE3Switchable::linearizeOplus()
63 {
64  g2o::VertexSE3* from = static_cast<g2o::VertexSE3*>(_vertices[0]);
65  g2o::VertexSE3* to = static_cast<g2o::VertexSE3*>(_vertices[1]);
66  const VertexSwitchLinear* vSwitch =
67  static_cast<const VertexSwitchLinear*>(_vertices[2]);
68 
69  Eigen::Isometry3d E;
70  const Eigen::Isometry3d& Xi = from->estimate();
71  const Eigen::Isometry3d& Xj = to->estimate();
72  const Eigen::Isometry3d& Z = _measurement;
73  g2o::internal::computeEdgeSE3Gradient(E, _jacobianOplus[0], _jacobianOplus[1],
74  Z, Xi, Xj);
75 
76  _jacobianOplus[0] *= vSwitch->estimate();
77  _jacobianOplus[1] *= vSwitch->estimate();
78 
79  // derivative w.r.t switch vertex
80  _jacobianOplus[2].setZero();
81  _jacobianOplus[2] = g2o::internal::toVectorMQT(E) * vSwitch->gradient();
82 }
