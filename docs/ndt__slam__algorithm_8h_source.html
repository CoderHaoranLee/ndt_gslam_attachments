185 NdtSlamAlgortihm::Pose NdtSlamAlgortihm::update(const Transform &motion,
186  const Covar &covariance,
187  const PointCloud &pcl,
188  const ros::Time &update_time)
189 {
190  // PROCESS OF ADDING TO POSE GRAPH
191  // 1. program is integrating measurements, transformations and covariances
192  // 2. robot moved enough
193  // 3 local map is added to mapper.
194  // 4 a) integrated covariance is copied to last known covariance var
195  // b) integrated transform is copied to last known transform var
196  // 5. first node is added to pose graph with current aggregated values (temp
197  // variables)
198  // 6. repeat 1.
199  // 7. execute 2. 3.
200  // 8. add new node to graph
201  // 9. add edge between last two nodes in graph with values saved in step 3
202  // 10. execute 4. and repeat process
203  if (!initialized_) {
204  running_window_->setOrigin(FrameType::Pose(0, 0, 0));
205  running_window_->setCellSize(cell_size_);
206  frame_temp_->setOrigin(FrameType::Pose(0, 0, 0));
207  frame_temp_->setCellSize(cell_size_);
208  position_.setIdentity();
209  last_matcher_pose_.setZero();
210  initialized_ = true;
211  // enters first scan as it is to running window
212  running_window_->enlarge(-win_radius_, -win_radius_, win_radius_,
213  win_radius_);
214  running_window_->initialize(pcl);
215  frame_temp_->initialize(pcl);
216  window_update_time_ = update_time;
217  window_seq_ = 0;
218  return eigt::getPoseFromTransform(position_);
219  }
220  unused_trans_ = unused_trans_ * motion;
221  if (!movedEnough(unused_trans_)) {
222  return eigt::getPoseFromTransform(position_ * motion);
223  }
224  unused_trans_.setIdentity();
225  FrameTypePtr local = FrameTypePtr(new FrameType());
226  PointCloud::Ptr pcl_out(new PointCloud());
227  local->setCellSize(cell_size_);
228  local->setOrigin(running_window_->getOrigin());
229  local->initialize(pcl);
230  // std::cout << *local << std::endl;
231  inc_matcher_.setInputTarget(running_window_);
232  inc_matcher_.setInputSource(local->getMeans());
233  // inc_matcher_.setInputSource(local);
234  inc_matcher_.align(
235  *pcl_out,
236  eigt::convertFromTransform(position_cumul_ * motion).cast<float>());
237  //
238  ROS_DEBUG_STREAM("[SLAM ALGORITHM]: incremental scanamtching converged:"
239  << inc_matcher_.hasConverged());
240  if (inc_matcher_.hasConverged()) {
241  // prepare transformation from successful scan registration
242  Transform registration = eigt::convertToTransform<double>(
243  inc_matcher_.getFinalTransformation().cast<double>());
244 
245  // GRAPH SLAM STUFF////////////////////////////////////////////
246  Pose matcher_pose = eigt::getPoseFromTransform(
247  eigt::getTransFromPose(running_window_->getOrigin()) * registration);
248 
249  // calculate pose increase with respect to last known pose in worls of
250  // scanmatching
251  Transform increase = eigt::getTransFromPose(last_matcher_pose_).inverse() *
252  eigt::getTransFromPose(matcher_pose);
253  std::cout << eigt::getAngle(increase)
254  << " dis: " << eigt::getDisplacement(increase) << std::endl
255  << std::endl;
256  if (eigt::getAngle(increase) < 0.01 &&
257  eigt::getDisplacement(increase) < 0.01) {
258  // increase = motion;
259  std::cout << "ODOM" << std::endl;
260  }
261 
262  last_matcher_pose_ = matcher_pose;
263 
264  // update and optimize graph
265  updateGraph(increase, Covar::Identity() * 0.01, pcl, update_time);
266 
267  // RUNNING WINDOW STUFF ///////////////////////////////////
268  // updating cumulative position. This position meassures
269  // transformation is
270  // relative to running window origin
271  position_cumul_ = registration;
272  local->transform(registration);
273  // merge in new data to running window
274  running_window_->mergeInTraced(*local, true, true);
275  // move running window only horizontaly or verticaly if needed
276  position_cumul_ = running_window_->move(registration);
277 
278  // info for output msg
279  window_update_time_ = update_time;
280  ++window_seq_;
281 
282  } else {
283  ROS_INFO_STREAM("[SLAM ALGORITHM]: unsucessful scanmatching-> using "
284  "odometry");
285  // scan-matching failed, use odometry
286  // do not merge in scan - avoiding problems whit bad local map
287  // position_ = position_ * motion;
288  updateGraph(motion, Covar::Identity() * 100, pcl, update_time);
289  }
290  // unused_trans_.setIdentity();
291  return eigt::getPoseFromTransform(position_);
292 }

294 void NdtSlamAlgortihm::updateGraph(const Transform &increase,
295  const Covar &covariance,
296  const PointCloud &new_scan,
297  const ros::Time &update_time)
298 {
299  transform_temp_ = transform_temp_ * increase;
300 
301  covar_temp_.addToCovar(covariance, increase);
302 
303  if (movedEnough(transform_temp_, covar_temp_)) {
304  ROS_INFO_STREAM("[SLAM ALGORITHM]: creating new frame");
305  // robot moved out of reasonable bounds for single temp
306  // frame
307  frames_.emplace_back(new FrameType());
308  frames_.back()->setCellSize(cell_size_);
309  frames_.back().swap(frame_temp_);
310  map_.addFrame(frames_.back(), update_time);
311  FrameTypeHolder frame_wrap(frames_.back());
312  size_t backup_id = last_node_id_;
313 
314  last_node_id_ =
315  opt_engine_->addPose(frames_.back()->getOrigin(), frame_wrap);
316  // add edge measurement only after first node was inserted
317  if (first_node_added_) {
318  opt_engine_->addLastConstrain(eigt::getPoseFromTransform(last_trans_),
319  last_covar_.covar_.inverse());
320  }
321  last_covar_ = covar_temp_;
322  last_trans_ = transform_temp_;
323  covar_temp_ = CovarianceWrapper();
324  transform_temp_.setIdentity();
325  first_node_added_ = true;
326  // pcl::visualizePcl<PointType>(running_window_->getMeans());
327  // find loop closures and optimize graph
328  Pose pre_opt_pose = opt_engine_->getPoseLocation(last_node_id_);
329  if (opt_engine_->tryLoopClose()) {
330  // optimize graph
331  if (opt_engine_->optimalizeIterationaly()) {
332  // optimization updates origins in grid through FrameHolder
333  // interface
334  map_.recalc(update_time);
335  }
336  }
337  // initialize new grid for next round
338  Pose after_opt_pose = opt_engine_->getPoseLocation(last_node_id_);
339 
340  position_ = eigt::getTransFromPose(after_opt_pose) * last_trans_;
341 
342  // pcl::visualizePcl<PointType>(frames_.back()->getMeans());
343 
344  frame_temp_->setOrigin(eigt::getPoseFromTransform(position_));
345  frame_temp_->setCellSize(cell_size_);
346  // frame_temp_->mergeInTraced(*new_scan, true, true);
347  frame_temp_->mergeInTraced(new_scan, frame_temp_->getOrigin(), true);
348  // std::cout << "last nodes pose in graph: "
349  // << opt_engine_->getPoseLocation(last_node_id_).transpose()
350  // << std::endl;
351  // pcl::visualizePcl<PointType>(frame_temp_->getMeans());
352 
353  } else {
354  // robot is still in bounds of reasonable error from
355  // incremental
356  // scan matching. We don't want to add to many nodes into
357  // pose graph
358 
359  position_ = position_ * increase;
360  PointCloud pcl_out;
361  pcl::transformPointCloud(new_scan, pcl_out,
362  eigt::convertFromTransform(transform_temp_));
363  frame_temp_->mergeInTraced(pcl_out, frame_temp_->getOrigin(), true);
364  }
365 
366  // std::cout << "position: " <<
367  // eigt::getPoseFromTransform(position_).transpose()
368  // << std::endl
369  // << std::endl;
370 }
