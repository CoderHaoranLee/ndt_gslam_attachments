153 template <class Optimizer>
154 bool RRRLoopProofer<Optimizer>::interClusterConsistent(IdVector& candidates,
155  IdVector& good_clusters,
156  IdVector& bad_clusters)
157 {
158  if (candidates.empty())
159  return true;
160 
161  VerticesIntPairSet activeLinks(gatherEdges(good_clusters));
162  auto temp = gatherEdges(candidates);
163  std::move(temp.begin(), temp.end(),
164  std::inserter(activeLinks, activeLinks.begin()));
165 
166  typename Optimizer::Result res;
167  res = optimizer_->optimize(activeLinks, n_iterations_);
168 
169  double activeChi2Graph = res.graph_error_;
170  size_t activeEdgeCount = res.edge_count_;
171 
172  double allLinksError = 0;
173  for (auto&& edge_err : res.loop_errors_) {
174  allLinksError += edge_err.second;
175  }
176 
177  if (activeChi2Graph < chi2(edge_dimention_ * activeEdgeCount) and
178  allLinksError < chi2(edge_dimention_ * activeLinks.size())) {
179  std::copy(candidates.begin(), candidates.end(),
180  std::back_inserter(good_clusters));
181  return true; // all done .. we found a consistent solution
182  } else {
183  // Find which cluster is causing the problems
184  // Iterate over everything
185  // Find the error for each cluster
186  // Sum cluster-wise
187  std::map<size_t, double> error_map;
188  for (const auto& edge_err : res.loop_errors_) {
189  error_map[clusterizer_.getClusterID(edge_err.first)] += edge_err.second;
190  }
191 
192  double min_CI = 0;
193  long rejectID = -1;
194  for (size_t cluster_id : candidates) {
195  double CI = error_map[cluster_id] /
196  chi2(edge_dimention_ *
197  clusterizer_.getCluster(cluster_id).loops_.size());
198  if (CI >= min_CI) // Just looking for the ones in candidates
199  {
200  min_CI = CI;
201  rejectID = static_cast<long>(cluster_id);
202  }
203  }
204  candidates.erase(candidates.begin() + rejectID);
205  bad_clusters.push_back(static_cast<size_t>(rejectID));
206 
207  return interClusterConsistent(candidates, good_clusters, bad_clusters);
208  }
209  return true;
210 }
