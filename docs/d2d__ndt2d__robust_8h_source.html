326 void D2DNormalDistributionsTransform2DRobust<
327  PointSource, PointTarget,
328  CellType>::computeTransformation(PclSource &output,
329  const Eigen::Matrix4f &guess)
330 {
331  // standard D2D match try for good guess estimations
332  d2d_.align(output, guess);
333  double score = proofTransform(d2d_.getFinalTransformation());
334  Eigen::Matrix4f first_trans;
335  // test if first d2d has return bad result
336  if (!(d2d_.hasConverged() && score > 0.7)) {
337  // bad result -> robust alignment needed
338  first_trans = d2d_.getFinalTransformation();
339  corr_est_.align(output, guess);
340  if (!corr_est_.hasConverged()) {
341  converged_ = false;
342  alignment_quality_ = 0;
343  final_transformation_.setIdentity();
344  return;
345  }
346  // second d2d -> precise alignment
347  d2d_.align(output, corr_est_.getFinalTransformation());
348  if (!d2d_.hasConverged()) { //||
350  //{
351  converged_ = false;
352  final_transformation_ = corr_est_.getFinalTransformation();
353  alignment_quality_ = 0;
354  return;
355  }
356  }
357  // score result
358  double score2 = proofTransform(d2d_.getFinalTransformation());
359  alignment_quality_ = score;
360  // robust alignment still not good enough
361  if (score2 < rejection_limit_) {
362  // Maybe at least first d2d got some reasonable result
363  if (score > 0.6) {
364  converged_ = true;
365  final_transformation_ = first_trans;
366  } else {
367  // everything is bad probably not the same place
368  converged_ = false;
369  final_transformation_ = d2d_.getFinalTransformation();
370  }
371  } else {
372  // we got good result by robust algorithm
373  converged_ = true;
374  alignment_quality_ = score2;
375  final_transformation_ = d2d_.getFinalTransformation();
376  }
377  // output cloud transform
378  transformPointCloud(*input_, output, final_transformation_);
379 }
