238 void D2DNormalDistributionsTransform2DRobust<
239  PointSource, PointTarget,
240  CellType>::computeTransformation(PclSource &output,
241  const Eigen::Matrix4f &guess)
242 {
243  // standard D2D match try for good guess estimations
244  d2d_.align(output, guess);
245  double score = proofTransform(d2d_.getFinalTransformation());
246  Eigen::Matrix4f first_trans;
247  // test if first d2d has return bad result
248  if (!(d2d_.hasConverged() && score > 0.7)) {
249  // bad result -> robust alignment needed
250  first_trans = d2d_.getFinalTransformation();
251  corr_est_.align(output, guess);
252  if (!corr_est_.hasConverged()) {
253  converged_ = false;
254  alignment_quality_ = 0;
255  final_transformation_.setIdentity();
256  return;
257  }
258  // second d2d -> precise alignment
259  d2d_.align(output, corr_est_.getFinalTransformation());
260  if (!d2d_.hasConverged()) { //||
262  //{
263  converged_ = false;
264  final_transformation_ = corr_est_.getFinalTransformation();
265  alignment_quality_ = 0;
266  return;
267  }
268  }
269  // score result
270  double score2 = proofTransform(d2d_.getFinalTransformation());
271  alignment_quality_ = score;
272  // robust alignment still not good enough
273  if (score2 < rejection_limit_) {
274  // Maybe at least first d2d got some reasonable result
275  if (score > 0.6) {
276  converged_ = true;
277  final_transformation_ = first_trans;
278  } else {
279  // everything is bad probably not the same place
280  converged_ = false;
281  final_transformation_ = d2d_.getFinalTransformation();
282  }
283  } else {
284  // we got good result by robust algorithm
285  converged_ = true;
286  alignment_quality_ = score2;
287  final_transformation_ = d2d_.getFinalTransformation();
288  }
289  // output cloud transform
290  transformPointCloud(*input_, output, final_transformation_);
291 }
